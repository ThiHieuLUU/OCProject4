487 def get_module_files(src_directory, blacklist, list_all=False):
488 """given a package directory return a list of all available python
489 module's files in the package and its subpackages
490  
491 :type src_directory: str
492 :param src_directory:
493 path of the directory corresponding to the package
494  
495 :type blacklist: list or tuple
496 :param blacklist: iterable
497 list of files or directories to ignore.
498  
499 :type list_all: bool
500 :param list_all:
501 get files from all paths, including ones without __init__.py
502  
503 :rtype: list
504 :return:
505 the list of all available python module's files in the package and
506 its subpackages
507 """
508 files = []
509 for directory, dirnames, filenames in os.walk(src_directory):
510 if directory in blacklist:
511 continue
512 _handle_blacklist(blacklist, dirnames, filenames)
513 # check for __init__.py
514 if not list_all and "__init__.py" not in filenames:
515 dirnames[:] = ()

518 if _is_python_file(filename):
519 src = os.path.join(directory, filename)

521 return files

524 def get_source_file(filename, include_no_ext=False):
525 """given a python module's file name return the matching source file
526 name (the filename will be returned identically if it's already an
527 absolute path to a python source file...)
528  
529 :type filename: str
530 :param filename: python module's file name
531  
532  
533 :raise NoSourceFile: if no source file exists on the file system
534  
535 :rtype: str
536 :return: the absolute path of the source file if it exists
537 """
538 filename = os.path.abspath(_path_from_filename(filename))
539 base, orig_ext = os.path.splitext(filename)
540 for ext in PY_SOURCE_EXTS:
541 source_path = "%s.%s" % (base, ext)
542 if os.path.exists(source_path):
543 return source_path
544 if include_no_ext and not orig_ext and os.path.exists(base):
545 return base
546 raise NoSourceFile(filename)

40 if name == "next":

118 def _instancecheck(cls, other):
119 wrapped = cls.__wrapped__
120 other_cls = other.__class__
121 is_instance_of = wrapped is other_cls or issubclass(other_cls, wrapped)
122 warnings.warn(
123 "%r is deprecated and slated for removal in astroid "
124 "2.0, use %r instead" % (cls.__class__.__name__, wrapped.__name__),
125 PendingDeprecationWarning,
126 stacklevel=2,
127 )
128 return is_instance_of

17 class BufferMapping(dict):
18 """
19 Dictionary that maps the name of the buffers to the
20 :class:`~prompt_toolkit.buffer.Buffer` instances.
21  
22 This mapping also keeps track of which buffer currently has the focus.
23 (Some methods receive a 'cli' parameter. This is useful for applications
24 where this `BufferMapping` is shared between several applications.)
25 """
26 def __init__(self, buffers=None, initial=DEFAULT_BUFFER):
27 assert buffers is None or isinstance(buffers, dict)
28  
29 # Start with an empty dict.
30 super(BufferMapping, self).__init__()
31  
32 # Add default buffers.
33 self.update({
34 # For the 'search' and 'system' buffers, 'returnable' is False, in
35 # order to block normal Enter/ControlC behaviour.
36 DEFAULT_BUFFER: Buffer(accept_action=AcceptAction.RETURN_DOCUMENT),
  • E501 Line too long (96 > 79 characters)
37 SEARCH_BUFFER: Buffer(history=InMemoryHistory(), accept_action=AcceptAction.IGNORE),
  • E501 Line too long (96 > 79 characters)
38 SYSTEM_BUFFER: Buffer(history=InMemoryHistory(), accept_action=AcceptAction.IGNORE),
39 DUMMY_BUFFER: Buffer(read_only=True),
40 })
41  
42 # Add received buffers.
43 if buffers is not None:
44 self.update(buffers)
45  
46 # Focus stack.
47 self.focus_stack = [initial or DEFAULT_BUFFER]
48  
49 def current(self, cli):
50 """
51 The active :class:`.Buffer`.
52 """
53 return self[self.focus_stack[-1]]
54  
55 def current_name(self, cli):
56 """
57 The name of the active :class:`.Buffer`.

62 """
63 Return the previously focussed :class:`.Buffer` or `None`.

66 try:
67 return self[self.focus_stack[-2]]

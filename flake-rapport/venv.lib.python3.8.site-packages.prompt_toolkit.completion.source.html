137 """

139 def doesnt_change_before_cursor(completion):
140 end = completion.text[:-completion.start_position]
141 return document.text_before_cursor.endswith(end)

143 completions2 = [c for c in completions if doesnt_change_before_cursor(c)]

148 return ''

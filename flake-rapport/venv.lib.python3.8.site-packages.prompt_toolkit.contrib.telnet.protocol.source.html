46 class TelnetProtocolParser(object):
47 """
48 Parser for the Telnet protocol.
49 Usage::
50  
51 def data_received(data):
52 print(data)
53  
54 def size_received(rows, columns):
55 print(rows, columns)
56  
57 p = TelnetProtocolParser(data_received, size_received)
58 p.feed(binary_data)
59 """
60 def __init__(self, data_received_callback, size_received_callback):
61 self.data_received_callback = data_received_callback
62 self.size_received_callback = size_received_callback
63  
64 self._parser = self._parse_coroutine()
65 self._parser.send(None)
66  
67 def received_data(self, data):
68 self.data_received_callback(data)
69  
70 def do_received(self, data):
71 """ Received telnet DO command. """
72 logger.info('DO %r', data)
73  
74 def dont_received(self, data):
75 """ Received telnet DONT command. """
76 logger.info('DONT %r', data)
77  
78 def will_received(self, data):
79 """ Received telnet WILL command. """
80 logger.info('WILL %r', data)
81  
82 def wont_received(self, data):
83 """ Received telnet WONT command. """
84 logger.info('WONT %r', data)
85  
86 def command_received(self, command, data):
87 if command == DO:
88 self.do_received(data)
89  
90 elif command == DONT:
91 self.dont_received(data)
92  
93 elif command == WILL:
94 self.will_received(data)
95  
96 elif command == WONT:
97 self.wont_received(data)
98  
99 else:
100 logger.info('command received %r %r', command, data)
101  
102 def naws(self, data):
103 """
104 Received NAWS. (Window dimensions.)
105 """
106 if len(data) == 4:
107 # NOTE: the first parameter of struct.unpack should be
108 # a 'str' object. Both on Py2/py3. This crashes on OSX
109 # otherwise.
110 columns, rows = struct.unpack(str('!HH'), data)
111 self.size_received_callback(rows, columns)
112 else:
113 logger.warning('Wrong number of NAWS bytes')
114  
115 def negotiate(self, data):
116 """
117 Got negotiate data.
118 """
119 command, payload = data[0:1], data[1:]
120 assert isinstance(command, bytes)
121  
122 if command == NAWS:
123 self.naws(payload)
124 else:
125 logger.info('Negotiate (%r got bytes)', len(data))

127 def _parse_coroutine(self):
128 """
129 Parser state machine.
130 Every 'yield' expression returns the next byte.
131 """
132 while True:
133 d = yield
134  
135 if d == int2byte(0):
136 pass # NOP
137  
138 # Go to state escaped.
139 elif d == IAC:
140 d2 = yield
141  
142 if d2 == IAC:
143 self.received_data(d2)
144  
145 # Handle simple commands.
146 elif d2 in (NOP, DM, BRK, IP, AO, AYT, EC, EL, GA):
147 self.command_received(d2, None)
148  
149 # Handle IAC-[DO/DONT/WILL/WONT] commands.
150 elif d2 in (DO, DONT, WILL, WONT):
151 d3 = yield
152 self.command_received(d2, d3)
153  
154 # Subnegotiation
155 elif d2 == SB:
156 # Consume everything until next IAC-SE
157 data = []
158  
159 while True:
160 d3 = yield
161  
162 if d3 == IAC:
163 d4 = yield
164 if d4 == SE:
165 break
166 else:
167 data.append(d4)
168 else:
169 data.append(d3)
170  
171 self.negotiate(b''.join(data))
172 else:
173 self.received_data(d)
174  
175 def feed(self, data):
176 """
177 Feed data to the parser.
178 """
179 assert isinstance(data, binary_type)
180 for b in iterbytes(data):
181 self._parser.send(int2byte(b))

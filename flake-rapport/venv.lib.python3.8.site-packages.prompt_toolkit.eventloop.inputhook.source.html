1 """
2 Similar to `PyOS_InputHook` of the Python API. Some eventloops can have an
3 inputhook to allow easy integration with other event loops.
4  
5 When the eventloop of prompt-toolkit is idle, it can call such a hook. This
6 hook can call another eventloop that runs for a short while, for instance to
7 keep a graphical user interface responsive.
8  
9 It's the responsibility of this hook to exit when there is input ready.
10 There are two ways to detect when input is ready:
11  
12 - Call the `input_is_ready` method periodically. Quit when this returns `True`.
13  
  • E501 Line too long (82 > 79 characters)
14 - Add the `fileno` as a watch to the external eventloop. Quit when file descriptor
15 becomes readable. (But don't read from it.)
16  
17 Note that this is not the same as checking for `sys.stdin.fileno()`. The
18 eventloop of prompt-toolkit allows thread-based executors, for example for
19 asynchronous autocompletion. When the completion for instance is ready, we
20 also want prompt-toolkit to gain control again in order to display that.
21  
  • E501 Line too long (81 > 79 characters)
22 An alternative to using input hooks, is to create a custom `EventLoop` class that
23 controls everything.
24 """

60 def call_inputhook(self, input_is_ready_func):
61 """
62 Call the inputhook. (Called by a prompt-toolkit eventloop.)
63 """
64 self._input_is_ready = input_is_ready_func
65  
66 # Start thread that activates this pipe when there is input to process.
67 def thread():
68 input_is_ready_func(wait=True)
69 os.write(self._w, b'x')
70  
71 threading.Thread(target=thread).start()
72  
73 # Call inputhook.
74 self.inputhook(self)
75  
76 # Flush the read end of the pipe.
77 try:
78 # Before calling 'os.read', call select.select. This is required
79 # when the gevent monkey patch has been applied. 'os.read' is never
80 # monkey patched and won't be cooperative, so that would block all
81 # other select() calls otherwise.
82 # See: http://www.gevent.org/gevent.os.html
83  
84 # Note: On Windows, this is apparently not an issue.
85 # However, if we would ever want to add a select call, it
86 # should use `windll.kernel32.WaitForMultipleObjects`,
87 # because `select.select` can't wait for a pipe on Windows.
88 if not is_windows():
89 select_fds([self._r], timeout=None)
90  
91 os.read(self._r, 1024)
92 except OSError:
93 # This happens when the window resizes and a SIGWINCH was received.
94 # We get 'Error: [Errno 4] Interrupted system call'
95 # Just ignore.
96 pass
97 self._input_is_ready = None
98  
99 def close(self):
100 """
101 Clean up resources.
102 """
103 if self._r:
104 os.close(self._r)
105 os.close(self._w)
106  
107 self._r = self._w = None

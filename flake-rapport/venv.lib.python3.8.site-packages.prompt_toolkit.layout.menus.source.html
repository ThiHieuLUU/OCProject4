153 def _trim_text(text, max_width):
154 """
155 Trim the text to `max_width`, append dots when the text is too long.
156 Returns (text, width) tuple.
157 """
158 width = get_cwidth(text)
159  
160 # When the text is too wide, trim it.
161 if width > max_width:
162 # When there are no double width characters, just use slice operation.
163 if len(text) == width:
164 trimmed_text = (text[:max(1, max_width-3)] + '...')[:max_width]
165 return trimmed_text, len(trimmed_text)
166  
167 # Otherwise, loop until we have the desired width. (Rather
168 # inefficient, but ok for now.)
169 else:
170 trimmed_text = ''
171 for c in text:
172 if get_cwidth(trimmed_text + c) <= max_width - 3:
173 trimmed_text += c
174 trimmed_text += '...'
175  
176 return (trimmed_text, get_cwidth(trimmed_text))
177 else:
178 return text, width

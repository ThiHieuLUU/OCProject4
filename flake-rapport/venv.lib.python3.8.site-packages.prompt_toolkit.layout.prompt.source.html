16 class DefaultPrompt(Processor):
17 """
18 Default prompt. This one shows the 'arg' and reverse search like
19 Bash/readline normally do.
20  
21 There are two ways to instantiate a ``DefaultPrompt``. For a prompt
22 with a static message, do for instance::
23  
24 prompt = DefaultPrompt.from_message('prompt> ')
25  
26 For a dynamic prompt, generated from a token list function::
27  
28 def get_tokens(cli):
29 return [(Token.A, 'text'), (Token.B, 'text2')]
30  
31 prompt = DefaultPrompt(get_tokens)
32 """

48 def apply_transformation(self, cli, document, lineno, source_to_display, tokens):
49 # Get text before cursor.
50 if cli.is_searching:
51 before = _get_isearch_tokens(cli)
52  
53 elif cli.input_processor.arg is not None:
54 before = _get_arg_tokens(cli)
55  
56 else:
57 before = self.get_tokens(cli)
58  
59 # Insert before buffer text.
60 shift_position = token_list_len(before)
61  
62 # Only show the prompt before the first line. For the following lines,
63 # only indent using spaces.
64 if lineno != 0:
65 before = [(Token.Prompt, ' ' * shift_position)]
66  
67 return Transformation(
68 tokens=before + tokens,
69 source_to_display=lambda i: i + shift_position,
70 display_to_source=lambda i: i - shift_position)

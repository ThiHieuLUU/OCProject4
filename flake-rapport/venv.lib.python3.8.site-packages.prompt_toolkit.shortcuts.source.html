1 """
2 Shortcuts for retrieving input from the user.
3  
4 If you are using this library for retrieving some input from the user (as a
5 pure Python replacement for GNU readline), probably for 90% of the use cases,
6 the :func:`.prompt` function is all you need. It's the easiest shortcut which
7 does a lot of the underlying work like creating a
8 :class:`~prompt_toolkit.interface.CommandLineInterface` instance for you.
9  
10 When is this not sufficient:
11 - When you want to have more complicated layouts (maybe with sidebars or
12 multiple toolbars. Or visibility of certain user interface controls
13 according to some conditions.)
14 - When you wish to have multiple input buffers. (If you would create an
15 editor like a Vi clone.)
16 - Something else that requires more customization than what is possible
17 with the parameters of `prompt`.
18  
19 In that case, study the code in this file and build your own
20 `CommandLineInterface` instance. It's not too complicated.

148 def _split_multiline_prompt(get_prompt_tokens):
149 """
150 Take a `get_prompt_tokens` function and return three new functions instead.
151 One that tells whether this prompt consists of multiple lines; one that
152 returns the tokens to be shown on the lines above the input; and another
153 one with the tokens to be shown at the first line of the input.
154 """
155 def has_before_tokens(cli):
156 for token, char in get_prompt_tokens(cli):
157 if '\n' in char:
158 return True
159 return False
160  
161 def before(cli):
162 result = []
163 found_nl = False
164 for token, char in reversed(explode_tokens(get_prompt_tokens(cli))):
165 if found_nl:
166 result.insert(0, (token, char))
167 elif char == '\n':
168 found_nl = True
169 return result
170  
171 def first_input_line(cli):
172 result = []
173 for token, char in reversed(explode_tokens(get_prompt_tokens(cli))):
174 if char == '\n':
175 break
176 else:
177 result.insert(0, (token, char))
178 return result
179  
180 return has_before_tokens, before, first_input_line

619 '''), exec_context)
620  
621 return exec_context['prompt_coro']()
622 else:
623 try:
624 with patch_context:
625 result = cli.run()
626  
627 if isinstance(result, Document): # Backwards-compatibility.
628 return result.text
629 return result
630 finally:
631 eventloop.close()

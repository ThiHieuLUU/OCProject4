106 def _get_closest_ansi_color(r, g, b, exclude=()):
107 """
108 Find closest ANSI color. Return it by name.
109  
110 :param r: Red (Between 0 and 255.)
111 :param g: Green (Between 0 and 255.)
112 :param b: Blue (Between 0 and 255.)
  • E501 Line too long (80 > 79 characters)
113 :param exclude: A tuple of color names to exclude. (E.g. ``('ansired', )``.)
114 """
115 assert isinstance(exclude, tuple)
116  
117 # When we have a bit of saturation, avoid the gray-like colors, otherwise,
118 # too often the distance to the gray color is less.
119 saturation = abs(r - g) + abs(g - b) + abs(b - r) # Between 0..510
120  
121 if saturation > 30:
122 exclude += ('ansilightgray', 'ansidarkgray', 'ansiwhite', 'ansiblack')
123  
124 # Take the closest color.
125 # (Thanks to Pygments for this part.)
126 distance = 257*257*3 # "infinity" (>distance from #000000 to #ffffff)
127 match = 'ansidefault'
128  
129 for name, (r2, g2, b2) in ANSI_COLORS_TO_RGB.items():
130 if name != 'ansidefault' and name not in exclude:
131 d = (r - r2) ** 2 + (g - g2) ** 2 + (b - b2) ** 2
132  
133 if d < distance:
134 match = name
135 distance = d
136  
137 return match

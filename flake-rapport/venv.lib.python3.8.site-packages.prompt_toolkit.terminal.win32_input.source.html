124 def read(self):
125 """
126 Return a list of `KeyPress` instances. It won't return anything when
127 there was nothing to read. (This function doesn't block.)
128  
129 http://msdn.microsoft.com/en-us/library/windows/desktop/ms684961(v=vs.85).aspx
130 """
131 max_count = 2048 # Max events to read at the same time.
132  
133 read = DWORD(0)
134 arrtype = INPUT_RECORD * max_count
135 input_records = arrtype()
136  
137 # Get next batch of input event.
138 windll.kernel32.ReadConsoleInputW(
139 self.handle, pointer(input_records), max_count, pointer(read))
140  
141 # First, get all the keys from the input buffer, in order to determine
142 # whether we should consider this a paste event or not.
143 all_keys = list(self._get_keys(read, input_records))
144  
145 if self.recognize_paste and self._is_paste(all_keys):
146 gen = iter(all_keys)
147 for k in gen:
148 # Pasting: if the current key consists of text or \n, turn it
149 # into a BracketedPaste.
150 data = []
151 while k and (isinstance(k.key, six.text_type) or
152 k.key == Keys.ControlJ):
153 data.append(k.data)
154 try:
155 k = next(gen)
156 except StopIteration:
157 k = None
158  
159 if data:
160 yield KeyPress(Keys.BracketedPaste, ''.join(data))
161 if k is not None:
162 yield k
163 else:
164 for k in all_keys:
165 yield k

167 def _get_keys(self, read, input_records):
168 """
169 Generator that yields `KeyPress` objects from the input records.
170 """
171 for i in range(read.value):
172 ir = input_records[i]
173  
174 # Get the right EventType from the EVENT_RECORD.
175 # (For some reason the Windows console application 'cmder'
176 # [http://gooseberrycreative.com/cmder/] can return '0' for
177 # ir.EventType. -- Just ignore that.)
178 if ir.EventType in EventTypes:
179 ev = getattr(ir.Event, EventTypes[ir.EventType])
180  
181 # Process if this is a key event. (We also have mouse, menu and
182 # focus events.)
183 if type(ev) == KEY_EVENT_RECORD and ev.KeyDown:
184 for key_press in self._event_to_key_presses(ev):
185 yield key_press
186  
187 elif type(ev) == MOUSE_EVENT_RECORD:
188 for key_press in self._handle_mouse(ev):
189 yield key_press

192 def _is_paste(keys):
193 """
194 Return `True` when we should consider this list of keys as a paste
195 event. Pasted text on windows will be turned into a
196 `Keys.BracketedPaste` event. (It's not 100% correct, but it is probably
197 the best possible way to detect pasting of text and handle that
198 correctly.)
199 """
200 # Consider paste when it contains at least one newline and at least one
201 # other character.
202 text_count = 0
203 newline_count = 0
204  
205 for k in keys:
206 if isinstance(k.key, six.text_type):
207 text_count += 1
208 if k.key == Keys.ControlJ:
209 newline_count += 1
210  
211 return newline_count >= 1 and text_count > 1

313 class raw_mode(object):
314 """
315 ::
316  
317 with raw_mode(stdin):
318 ''' the windows terminal is now in 'raw' mode. '''
319  
320 The ``fileno`` attribute is ignored. This is to be compatble with the
321 `raw_input` method of `.vt100_input`.
322 """

324 self.handle = windll.kernel32.GetStdHandle(STD_INPUT_HANDLE)

326 def __enter__(self):
327 # Remember original mode.
328 original_mode = DWORD()
329 windll.kernel32.GetConsoleMode(self.handle, pointer(original_mode))
330 self.original_mode = original_mode
331  
332 self._patch()
333  
334 def _patch(self):
335 # Set raw
336 ENABLE_ECHO_INPUT = 0x0004
337 ENABLE_LINE_INPUT = 0x0002
338 ENABLE_PROCESSED_INPUT = 0x0001
339  
340 windll.kernel32.SetConsoleMode(
341 self.handle, self.original_mode.value &
342 ~(ENABLE_ECHO_INPUT | ENABLE_LINE_INPUT | ENABLE_PROCESSED_INPUT))
343  
344 def __exit__(self, *a, **kw):
345 # Restore original mode
346 windll.kernel32.SetConsoleMode(self.handle, self.original_mode)

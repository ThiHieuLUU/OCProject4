143 class _CharSizesCache(dict):
144 """
145 Cache for wcwidth sizes.
146 """
147 def __missing__(self, string):
148 # Note: We use the `max(0, ...` because some non printable control
149 # characters, like e.g. Ctrl-underscore get a -1 wcwidth value.
150 # It can be possible that these characters end up in the input
151 # text.
152 if len(string) == 1:
153 result = max(0, wcwidth(string))
154 else:
155 result = sum(max(0, wcwidth(c)) for c in string)
156  
157 # Cache for short strings.
158 # (It's hard to tell what we can consider short...)
159 if len(string) < 256:
160 self[string] = result
161  
162 return result

49 def table_lines_from_stats(stats, _, columns):
50 """get values listed in <columns> from <stats> and <old_stats>,
51 and return a formated list of values, designed to be given to a
52 ureport.Table object
53 """
54 lines = []
55 for m_type in columns:
56 new = stats[m_type]
57 new = "%.3f" % new if isinstance(new, float) else str(new)
58 lines += (m_type.replace("_", " "), new, "NC", "NC")
59 return lines

113 def check_consistency(self):
114 """Check the consistency of msgid.
115  
116 msg ids for a checker should be a string of len 4, where the two first
117 characters are the checker id and the two last the msg id in this
118 checker.
119  
120 :raises InvalidMessageError: If the checker id in the messages are not
121 always the same. """
122 checker_id = None
123 existing_ids = []
124 for message in self.messages:
125 if checker_id is not None and checker_id != message.msgid[1:3]:
126 error_msg = "Inconsistent checker part in message id "

128 message.msgid, checker_id=checker_id
129 )
130 error_msg += "because we already had {existing_ids}).".format(
131 existing_ids=existing_ids
132 )
133 raise InvalidMessageError(error_msg)

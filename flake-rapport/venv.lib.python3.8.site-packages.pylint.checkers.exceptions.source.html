53 def _annotated_unpack_infer(stmt, context=None):
54 """
55 Recursively generate nodes inferred by the given statement.
56 If the inferred value is a list or a tuple, recurse on the elements.
57 Returns an iterator which yields tuples in the format
58 ('original node', 'inferred node').
59 """
60 if isinstance(stmt, (astroid.List, astroid.Tuple)):
61 for elt in stmt.elts:
62 inferred = utils.safe_infer(elt)
63 if inferred and inferred is not astroid.Uninferable:
64 yield elt, inferred
65 return
66 for inferred in stmt.infer(context):
67 if inferred is astroid.Uninferable:
68 continue
69 yield stmt, inferred

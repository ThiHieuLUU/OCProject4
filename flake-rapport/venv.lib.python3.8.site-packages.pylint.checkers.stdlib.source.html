53 def _check_mode_str(mode):
54 # check type
55 if not isinstance(mode, str):
56 return False
57 # check syntax
58 modes = set(mode)
59 _mode = "rwatb+Ux"
60 creating = "x" in modes
61 if modes - set(_mode) or len(mode) > len(modes):
62 return False
63 # check logic
64 reading = "r" in modes
65 writing = "w" in modes
66 appending = "a" in modes
67 text = "t" in modes
68 binary = "b" in modes
69 if "U" in modes:
70 if writing or appending or creating:
71 return False
72 reading = True
73 if text and binary:
74 return False
75 total = reading + writing + appending + creating
76 if total > 1:
77 return False
78 if not (reading or writing or appending or creating):
79 return False
80 return True

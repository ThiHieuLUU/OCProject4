8 """Looks for try/except statements with too much code in the try clause."""

10 from astroid.node_classes import For, If, While, With

12 from pylint import checkers, interfaces

15 class BroadTryClauseChecker(checkers.BaseChecker):
16 """Checks for try clauses with too many lines.
17  
18 According to PEP 8, ``try`` clauses shall contain the absolute minimum
19 amount of code. This checker enforces a maximum number of statements within
20 ``try`` clauses.
21  
22 """
23  
24 __implements__ = interfaces.IAstroidChecker
25  
26 # configuration section name
27 name = "broad_try_clause"
28 msgs = {
29 "W0717": (
30 "%s",
31 "too-many-try-statements",
32 "Try clause contains too many statements.",
33 )
34 }
35  
36 priority = -2
37 options = (
38 (
39 "max-try-statements",
40 {
41 "default": 1,
42 "type": "int",
43 "metavar": "<int>",
44 "help": "Maximum number of statements allowed in a try clause",
45 },
46 ),
47 )
48  
49 def _count_statements(self, try_node):
50 statement_count = len(try_node.body)
51  
52 for body_node in try_node.body:
53 if isinstance(body_node, (For, If, While, With)):
54 statement_count += self._count_statements(body_node)
55  
56 return statement_count
57  
58 def visit_tryexcept(self, node):
59 try_clause_statements = self._count_statements(node)
60 if try_clause_statements > self.config.max_try_statements:
  • E501 Line too long (82 > 79 characters)
61 msg = "try clause contains {} statements, expected at most {}".format(
62 try_clause_statements, self.config.max_try_statements
63 )
64 self.add_message(
65 "too-many-try-statements", node.lineno, node=node, args=msg
66 )
67  
68 def visit_tryfinally(self, node):
69 self.visit_tryexcept(node)

31 class Interface:
32 """Base class for interfaces."""
33  
34 @classmethod
35 def is_implemented_by(cls, instance):
36 return implements(instance, cls)
37  
38  
39 def implements(obj, interface):
40 """Return true if the give object (maybe an instance or class) implements
41 the interface.
42 """
43 kimplements = getattr(obj, "__implements__", ())
44 if not isinstance(kimplements, (list, tuple)):
45 kimplements = (kimplements,)
46 for implementedinterface in kimplements:
47 if issubclass(implementedinterface, interface):
48 return True
49 return False

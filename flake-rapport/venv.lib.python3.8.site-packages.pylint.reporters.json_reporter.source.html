13 """JSON reporter"""
14 import json
15 import sys
16  
17 from pylint.interfaces import IReporter
18 from pylint.reporters.base_reporter import BaseReporter
19  
20  
21 class JSONReporter(BaseReporter):
22 """Report messages and layouts in JSON."""
23  
24 __implements__ = IReporter
25 name = "json"
26 extension = "json"
27  
28 def __init__(self, output=None):
29 BaseReporter.__init__(self, output or sys.stdout)
30 self.messages = []
31  
32 def handle_message(self, msg):
33 """Manage message of different type and in the context of path."""
34 self.messages.append(
35 {
36 "type": msg.category,
37 "module": msg.module,
38 "obj": msg.obj,
39 "line": msg.line,
40 "column": msg.column,
41 "path": msg.path,
42 "symbol": msg.symbol,
43 "message": msg.msg or "",
44 "message-id": msg.msg_id,
45 }
46 )
47  
48 def display_messages(self, layout):
49 """Launch layouts display"""
50 print(json.dumps(self.messages, indent=4), file=self.out)
51  
52 def display_reports(self, layout):
53 """Don't do anything in this reporter."""
54  
55 def _display(self, layout):
56 """Do nothing."""
57  
58  
59 def register(linter):
60 """Register the reporter classes with the linter."""
61 linter.register_reporter(JSONReporter)

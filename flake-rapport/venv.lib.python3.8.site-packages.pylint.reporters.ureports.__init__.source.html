10 """Universal report objects and some formatting drivers.
11  
12 A way to create simple reports using python objects, primarily designed to be
13 formatted as text and html.
14 """
15 import os
16 import sys
17 from io import StringIO
18  
19  
20 class BaseWriter:
21 """base class for ureport writers"""
22  
23 def format(self, layout, stream=None, encoding=None):
24 """format and write the given layout into the stream object
25  
26 unicode policy: unicode strings may be found in the layout;
27 try to call stream.write with it, but give it back encoded using
28 the given encoding if it fails
29 """
30 if stream is None:
31 stream = sys.stdout
32 if not encoding:
33 encoding = getattr(stream, "encoding", "UTF-8")
34 self.encoding = encoding or "UTF-8"
35 self.out = stream
36 self.begin_format()
37 layout.accept(self)
38 self.end_format()
39  
40 def format_children(self, layout):
41 """recurse on the layout children and call their accept method
42 (see the Visitor pattern)
43 """
44 for child in getattr(layout, "children", ()):
45 child.accept(self)
46  
47 def writeln(self, string=""):
48 """write a line in the output buffer"""
49 self.write(string + os.linesep)
50  
51 def write(self, string):
52 """write a string in the output buffer"""
53 self.out.write(string)
54  
55 def begin_format(self):
56 """begin to format a layout"""
57 self.section = 0
58  
59 def end_format(self):
60 """finished to format a layout"""
61  
62 def get_table_content(self, table):
63 """trick to get table content without actually writing it
64  
65 return an aligned list of lists containing table cells values as string
66 """
67 result = [[]]
68 cols = table.cols
69 for cell in self.compute_content(table):
70 if cols == 0:
71 result.append([])
72 cols = table.cols
73 cols -= 1
74 result[-1].append(cell)
75 # fill missing cells
76 while len(result[-1]) < cols:
77 result[-1].append("")
78 return result

80 def compute_content(self, layout):
81 """trick to compute the formatting of children layout before actually
82 writing it
83  
84 return an iterator on strings (one for each child element)
85 """
86 # Patch the underlying output stream with a fresh-generated stream,
87 # which is used to store a temporary representation of a child
88 # node.
89 out = self.out
90 try:
91 for child in layout.children:
92 stream = StringIO()
93 self.out = stream
94 child.accept(self)
95 yield stream.getvalue()
96 finally:
97 self.out = out

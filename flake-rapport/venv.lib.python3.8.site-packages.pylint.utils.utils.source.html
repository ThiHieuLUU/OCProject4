365 def _format_option_value(optdict, value):
366 """return the user input's value from a 'compiled' value"""
367 if isinstance(value, (list, tuple)):
368 value = ",".join(_format_option_value(optdict, item) for item in value)
369 elif isinstance(value, dict):

373 value = value.pattern
374 elif optdict.get("type") == "yn":
375 value = "yes" if value else "no"
376 elif isinstance(value, str) and value.isspace():
377 value = "'%s'" % value
